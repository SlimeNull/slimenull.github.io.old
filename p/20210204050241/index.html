def printrst(resp):

            print(" <0> Result:")
            if "translation" in respJson:

                print("       Basic:")

                print("       From web:")

        else:

def runcmd(cmd):

    if cmdname == '$':

        advancedmode = not advancedmode
        print(" <$> AdvancedMode: %s" % advancedmode)
